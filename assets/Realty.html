# delete outliers
def outliers(df, column, quantiles=(0.15,0.85)):
    """
      Removes outliers from dataframe
    """
    min_val = df[column].quantile(quantiles[0])
    max_val = df[column].quantile(quantiles[1])
    f = lambda x: min_val <= x <= max_val
    #return df[np.abs(df[column]-df[column].mean())<=(3*df[column].std())]
    return df[df[column].apply(f)]
